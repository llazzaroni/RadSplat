3) step: now the datasets are ready. Train the ensembles of nerf models

srun -A ls_polle -p gpuhe.4h --ntasks=1 --mem-per-cpu=4G --cpus-per-task=4 --gpus=nvidia_geforce_rtx_4090:1 --time=10:00:00 --pty bash

DATA_ROOT="/cluster/scratch/rbollati/new_exp/360_v2"
OUT_ROOT="/cluster/scratch/rbollati/new_exp"

NUM_ENSEMBLE=5
MAX_ITERS_RUN1=2500
MAX_ITERS_OTHER=2000
SAVE_EVERY=200
EVAL_EVERY=200

for dataset_dir in "${DATA_ROOT}"/*_sparse_vggt; do
  [[ -d "${dataset_dir}" ]] || continue

  dataset_name="$(basename "${dataset_dir}")"
  scene_name="${dataset_name%_sparse_vggt}"

  model_root="${OUT_ROOT}/models_${scene_name}"
  mkdir -p "${model_root}"

  for i in $(seq 1 "${NUM_ENSEMBLE}"); do
    run_dir="${model_root}/nerf_ensemble_${i}"
    mkdir -p "${run_dir}"
    cd "${run_dir}"

    if [[ "${i}" -eq 1 ]]; then
      ns-train depth-nerfacto \
        --vis tensorboard \
        --experiment-name "ensemble_${i}" \
        --timestamp "${scene_name}" \
        --steps-per-save "${SAVE_EVERY}" \
        --steps-per-eval-all-images "${EVAL_EVERY}" \
        --steps-per-eval-image 0 \
        --max-num-iterations "${MAX_ITERS_RUN1}" \
        --save-only-latest-checkpoint False \
        --logging.steps-per-log 100 \
        --logging.profiler pytorch \
        colmap \
        --downscale-factor 1 \
        --colmap-path "${dataset_dir}/sparse" \
        --images-path "${dataset_dir}/images" \
        --depths-path "${dataset_dir}/depths_vggt"
    else
      ns-train depth-nerfacto \
        --vis tensorboard \
        --experiment-name "ensemble_${i}" \
        --timestamp "${scene_name}" \
        --steps-per-save "${SAVE_EVERY}" \
        --steps-per-eval-all-images 100000000 \
        --steps-per-eval-image 0 \
        --max-num-iterations "${MAX_ITERS_OTHER}" \
        --save-only-latest-checkpoint False \
        --logging.steps-per-log 100 \
        --logging.profiler pytorch \
        colmap \
        --downscale-factor 1 \
        --colmap-path "${dataset_dir}/sparse" \
        --images-path "${dataset_dir}/images" \
        --depths-path "${dataset_dir}/depths_vggt"
    fi

    cd - >/dev/null
  done
done

import json
from pathlib import Path
import matplotlib.pyplot as plt

files = {
    "depth-nerfacto": Path("/cluster/scratch/rbollati/new_exp/models_stump/depth-nerfacto/outputs/depth-nerfacto/depth-nerfacto/stump/eval_all_images.jsonl"),
    "nerfacto": Path("/cluster/scratch/rbollati/new_exp/models_stump/nerfacto/outputs/nerfacto/nerfacto/stump/eval_all_images.jsonl"),
}

metrics = ["psnr", "ssim", "lpips"]

series = {}
for label, path in files.items():
    rows = []
    with path.open() as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            rows.append(json.loads(line))
    rows.sort(key=lambda x: x["step"])
    series[label] = rows

fig, axes = plt.subplots(1, 3, figsize=(15, 4))

for ax, metric in zip(axes, metrics):
    for label, rows in series.items():
        xs = [r["step"] for r in rows if metric in r]
        ys = [r[metric] for r in rows if metric in r]
        ax.plot(xs, ys, marker="o", label=label)
    ax.set_title(metric)
    ax.set_xlabel("step")
    ax.set_ylabel(metric)
    ax.grid(True, alpha=0.3)
    ax.legend()

plt.tight_layout()
plt.savefig("/cluster/scratch/rbollati/new_exp/stump_nerf_compare.png", dpi=200, bbox_inches="tight")
