python - <<'PY'
import sqlite3
from pathlib import Path

sub = Path("/cluster/scratch/rbollati/dataset/treehill_sparse")  # change per dataset
txt_in = sub / "sparse/pose_txt"
txt_out = sub / "sparse/pose_txt_dbids"
db_path = sub / "database.db"

pose_by_name = {}
lines = (txt_in / "images.txt").read_text().splitlines()
i = 0
while i < len(lines):
    line = lines[i].strip()
    if not line or line.startswith("#"):
        i += 1
        continue
    parts = line.split()
    name = parts[9]
    pose_by_name[name] = parts[1:9]  # qw qx qy qz tx ty tz cam_id
    i += 2

con = sqlite3.connect(db_path)
rows = con.execute("SELECT image_id, name FROM images").fetchall()
con.close()

(txt_out / "cameras.txt").write_text((txt_in / "cameras.txt").read_text())

out = [
    "# Image list with two lines of data per image:",
    "# IMAGE_ID, QW, QX, QY, QZ, TX, TY, TZ, CAMERA_ID, IMAGE_NAME",
    "# POINTS2D[] as (X, Y, POINT3D_ID)",
]
for image_id, name in sorted(rows):
    qw,qx,qy,qz,tx,ty,tz,cam_id = pose_by_name[name]
    out.append(f"{image_id} {qw} {qx} {qy} {qz} {tx} {ty} {tz} {cam_id} {name}")
    out.append("")
(txt_out / "images.txt").write_text("\n".join(out) + "\n")
(txt_out / "points3D.txt").write_text(
    "# 3D point list with one line of data per point:\n"
    "# POINT3D_ID, X, Y, Z, R, G, B, ERROR, TRACK[] as (IMAGE_ID, POINT2D_IDX)\n"
)
print("pose_txt_dbids written")
PY

2) run nerf_step; this produces the tensor of positions and colors to initialize gaussian splatting

python nerf_step.py \
    --nerf-folder NERF_MODEL \
    --output-name POSITION_TENSOR_OUTPUT_NAME \
    --sampling-size SAMPLING_SIZE \
    --ray-sampling-strategy RAY_SAMPLING_STRATEGY \
    --percentage-random PERCENTAGE_RANDOM

NERF_MODEL is: nerf_folder/outputs/EXPERIMENT_NAME/nerfacto/EXPERIMENT_NAME/config.yml
POSITION_TENSOR_OUTPUT_NAME is: EXPERIMENT_DIR/ray_sample.pt
SAMPLING_SIZE: 1_000_000 for standard RadSplat
RAY_SAMPLING_STRATEGY: "random" for standard RadSplat; potentially "patches" for minimal small upgrade

3) generate the new images

python regen_img.py \
    --input-dataset DATA_DIR \
    --output-dataset OUT_DATA_DIR \
    --nerf-folder NERF_MODEL

What regen_img.py does:
- Copies DATA_DIR into OUT_DATA_DIR
- Removes image files from images, images_2, images_4, images_8 in OUT_DATA_DIR
- Re-renders images from NeRF camera poses loaded from NERF_MODEL
- Saves regenerated images to images and regenerated downscaled versions to images_2/images_4/images_8

import json, sys, pathlib
run_dir = pathlib.Path(sys.argv[1])
step = int(sys.argv[2])
p = run_dir / "gsplat_stats.json"
arr = json.load(open(p))
if not isinstance(arr, list) or len(arr) == 0:
    raise SystemExit(f"No stats in {p}")

# Exact step if present, else fallback to index=step//100 (default logging interval)
row = next((x for x in arr if isinstance(x, dict) and x.get("step") == step), None)
if row is None:
    idx = step // 100
    if idx >= len(arr):
        raise SystemExit(f"Step {step} not found, max approx step is {(len(arr)-1)*100}")
    row = arr[idx]

print(f"metrics at step {step}:")
